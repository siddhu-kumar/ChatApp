const express = require('express');
const app = express();

// Middleware to handle CORS manually
app.use((req, res, next) => {
  // Allow specific origin (replace with your front-end domain)
  const allowedOrigin = 'http://localhost:3000';  // Frontend domain

  // Check if the origin of the request is allowed
  if (req.headers.origin === allowedOrigin) {
    // Allow the origin if it matches
    res.header('Access-Control-Allow-Origin', allowedOrigin);
  } else {
    // Optionally, block other origins
    res.status(403).send('Origin not allowed');
    return;
  }

  // Allow specific HTTP methods (GET, POST, PUT, DELETE, etc.)
  res.header('Access-Control-Allow-Methods', 'GET, POST, PUT, DELETE, OPTIONS');

  // Allow specific headers in the request (e.g., Content-Type, Authorization)
  res.header('Access-Control-Allow-Headers', 'Content-Type, Authorization');

  // Allow credentials (cookies, authentication headers, etc.)
  res.header('Access-Control-Allow-Credentials', 'true');

  // Handle OPTIONS requests (preflight)
  if (req.method === 'OPTIONS') {
    return res.status(204).end();  // Respond with no content for OPTIONS request
  }

  // Proceed to the next middleware or route handler
  next();
});

// Sample route
app.get('/api/data', (req, res) => {
  res.json({ message: 'Hello from the server!' });
});

// Start the server
const port = 5000;
app.listen(port, () => {
  console.log(`Server running on http://localhost:${port}`);
});
